def visualize(datasets):
    "Plot the histogram"
    data = pd.read_csv(datasets)
    x = data.min()
    y = data.max()
    z = data.mean()
    plt.plot(x)
    plt.plot(y)
    plt.plot(z)
    plt.title('assignment graph')
    plt.plot(x, label='min')
    plt.plot(y, label='max')
    plt.plot(z, label='mean')
    plt.legend()
    plt.show()
    print (data.describe())
